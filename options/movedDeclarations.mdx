### Input / Output

This example showcases how `Moved Declarations` transforms the code. Try it out by changing the input code and see changes apply in real-time.

```js title="Input.js" lines interactive-mode="obfuscate"
// Input.js
function getAreaOfCircle(radius) {
  var pi = Math.PI;
  var radiusSquared = Math.pow(radius, 2);
  var area = pi * radiusSquared;

  return area;
}

console.log(getAreaOfCircle(3)); // 28.274333882308138

// Output.js
function PPA177Y(PPA177Y, iD9NBe, g0C37A3, eE11tuf) {
  iD9NBe = Math.PI;
  g0C37A3 = Math.pow(PPA177Y, 2);
  eE11tuf = iD9NBe * g0C37A3;
  return eE11tuf;
}
console.log(PPA177Y(3));
```

### Usage Example

The provided code example will obfuscate the file `input.js` and write the output to a file named `output.js`.

```js title="Usage Example" lines
import JSConfuser from "js-confuser";
import {readFileSync, writeFileSync} from "fs";

// Read input code
const sourceCode = readFileSync("input.js", "utf8");
const options = {
  target: 'browser',
  movedDeclarations: true,
};

JSConfuser.obfuscate(sourceCode, options).then((result)=>{
  // Write output code
  writeFileSync("output.js", result.code);
}).catch(err=>{
  // Error occurred
  console.error(err);  
});
```
