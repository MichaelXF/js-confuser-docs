#### Input / Output

This example showcases how `Flatten` transforms the code. Try it out by changing the input code and see changes apply in real-time.

```js title="Input.js" lines interactive-mode="obfuscate"
// Input.js
(function () {
  var stringToPrint = 'Hello World';
  var timesPrinted = 0;

  function printString() {
    timesPrinted++;
    console.log(stringToPrint);
  }

  printString(); // "Hello World"
})();

// Output.js
function Kh0SB5U([], Kh0SB5U) {
  var QTLXFv = 'Hello World',
    BS2DWlx = 0;

    var qCL5ohY = {

        return BS2DWlx;
      },

        BS2DWlx = dkivS0;
      },
      get Vb9YZ2S() {
        return QTLXFv;
      },
    };
    return Kh0SB5U.VN5FhW9(dkivS0, qCL5ohY);
  }
  dkivS0();
}
function QTLXFv([], Kh0SB5U) {
  Kh0SB5U.Tni5fwT++;
  console.log(Kh0SB5U.Vb9YZ2S);
}
```

#### Custom Implementation
###### `options.flatten(fnName)`

Control which functions are changed. Returns a `boolean`.
| Parameter | Type | Description |
| --- | --- | --- |
| `fnName` | `string` | The function name proposed to be changed. |


---

#### Usage Example

The provided code example will obfuscate the file `input.js` and write the output to a file named `output.js`.

```js title="Usage Example" lines
import JSConfuser from "js-confuser";
import {readFileSync, writeFileSync} from "fs";

// Read input code
const sourceCode = readFileSync("input.js", "utf8");
const options = {
  target: 'browser',
  flatten: true,
};

JSConfuser.obfuscate(sourceCode, options).then((result)=>{
  // Write output code
  writeFileSync("output.js", result.code);
}).catch(err=>{
  // Error occurred
  console.error(err);  
});
```
