### Input / Output

This example showcases how `Calculator` transforms the code. Try it out by changing the input code and see changes apply in real-time.

```js title="Input.js" lines interactive-mode="obfuscate"
// Input.js
var result = 9 + 3 * 2;
console.log(result);

// Output.js
function WrC1bId(WrC1bId, uRSYZk, r8iMsV) {
  switch (WrC1bId) {
    case 'crP9Cl':
      return uRSYZk * r8iMsV;
  }
}
var uRSYZk = 9 + WrC1bId('crP9Cl', 3, 2);
console.log(uRSYZk);
```

### Usage Example

The provided code example will obfuscate the file `input.js` and write the output to a file named `output.js`.

```js title="Usage Example" lines
import JSConfuser from "js-confuser";
import {readFileSync, writeFileSync} from "fs";

// Read input code
const sourceCode = readFileSync("input.js", "utf8");
const options = {
  target: 'browser',
  calculator: true,
};

JSConfuser.obfuscate(sourceCode, options).then((result)=>{
  // Write output code
  writeFileSync("output.js", result.code);
}).catch(err=>{
  // Error occurred
  console.error(err);  
});
```
